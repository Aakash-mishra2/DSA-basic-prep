   int i = 0, count = 0;
    for (; input[i] != '\0'; i++)
    {
        if (input[i] == ' ')
        {
            count++;
        }
    }

    int ne = i + 2 * (count);
    while (ne > i)
    {
        if (input[i] == ' ')
        {
            input[ne--] = '0';
            input[ne--] = '2';
            input[ne--] = '%';
            i--;
        }   
        else
        {
            input[ne--] = input[i--];
        }   
    }
    return input;
